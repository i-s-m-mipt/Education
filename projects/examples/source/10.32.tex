\textbf{Strassen Fast Multiplication Algorithm}

\bigskip

Let $A = $ 
\begin{Vmatrix}
\;A_{11} & A_{12}\; \\
\;A_{21} & A_{22}\;
\end{Vmatrix}, 
$B = $ 
\begin{Vmatrix}
\;B_{11} & B_{12}\; \\
\;B_{21} & B_{22}\;
\end{Vmatrix}, 
where $A,\,B \in M(N \times N)$ and $A_{ij},\,B_{ij} \in M(\frac{N}{2} \times \frac{N}{2})$ for $i,\,j \in \{\,1,\,2\,\}$

\bigskip

Hence, $C = A \times B = $
\begin{Vmatrix}
\;A_{11} \times B_{11} + A_{12} \times B_{21} & A_{11} \times B_{12} + A_{12} \times B_{22}\; \\
\;A_{21} \times B_{11} + A_{22} \times B_{21} & A_{21} \times B_{12} + A_{22} \times B_{22}\;
\end{Vmatrix} 
with time complexity $O(N^3)$

\bigskip

Let $C_1 = (A_{11} + A_{22}) \times (B_{11} + B_{22})$

\bigskip

Let $C_2 = (A_{11} + A_{12}) \times B_{22}$, $C_3 = A_{11} \times (B_{12} - B_{22})$

\bigskip

Let $C_4 = (A_{21} + A_{22}) \times B_{11}$, $C_5 = A_{22} \times (B_{21} - B_{11})$

\bigskip

Let $C_6 = (A_{21} - A_{11}) \times (B_{11} + B_{12})$

\bigskip

Let $C_7 = (A_{12} - A_{22}) \times (B_{21} + B_{22})$

\bigskip

Hence, $C = A \times B = $
\begin{Vmatrix}
\; C_1 - C_2 + C_5 + C_7 & C_2 + C_3\; \\
\; C_4 + C_5 & C_1 + C_3 - C_4 + C_6\;
\end{Vmatrix} 
with time complexity $O(N^{\log_{2}7})\approx O(N^{2.807})$

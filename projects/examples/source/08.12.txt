// content : Karatsuba Fast Multiplication Algorithm

/////////////////////////////////////////////////////////////////////////

x = x2 * 10 ^ n + x1, where len(x2) = len(x1)

y = y2 * 10 ^ n + y1, where len(y2) = len(y1)

/////////////////////////////////////////////////////////////////////////

x * y = x2 * y2 * 10 ^ 2n + (x2 * y1 + x1 * y2) * 10 ^ n + x1 * y1

/////////////////////////////////////////////////////////////////////////

a = x2 * y2, b = x1 * y1, c = (x2 + x1) * (y2 + y1)

/////////////////////////////////////////////////////////////////////////

x2 * y1 + x1 * y2 = (x2 + x1) * (y2 + y1) - x1 * y1 - x2 * y2 = c - b - a

/////////////////////////////////////////////////////////////////////////

x * y = a * 10 ^ 2n + (c - b - a) * 10 ^ n + b
